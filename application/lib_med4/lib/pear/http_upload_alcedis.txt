function _evalValidExtensions()
{
    $exts = $this->_extensions_check;
    settype($exts, 'array');

    if ($this->_extensions_mode == 'deny') {
        if (in_array(strtolower($this->getProp('ext')), $exts)) {
            return false;
        }
    // mode == 'accept'
    } else {
        if (!in_array(strtolower($this->getProp('ext')), $exts)) {
            return false;
        }
    }

    return true;
}
